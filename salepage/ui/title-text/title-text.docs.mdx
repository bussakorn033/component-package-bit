import { TitleText } from './title-text';

A component that does something special and renders text in a div.

### Component usage
```js
<TitleText
    tagColor={'#1DCAB4'}
    color={'#2A2C2D'}
    fontSize={'2rem'}
    fontWeight={'bold'}
    textTransform={'uppercase'}
>
    hello world!
</TitleText>
```

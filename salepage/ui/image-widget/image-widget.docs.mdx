import { ImageWidget } from './image-widget';

A component that does something special and renders text in a div.

### Component usage
```js
<ImageWidget
    link = {'#'}
    src = {'https://picsum.photos/0/0'}
    alt = {'ImageWidget'}
    objectFit = {'contain'}
/>
```
